### **🚀 Priority 3: Temporary Storage & Cleanup**

#### **3.1 Automated Cleanup Cron Job**

**File:** `app/api/cron/cleanup-unpurchased/route.ts`

```typescript
export async function GET() {
  try {
    const fortyEightHoursAgo = new Date(Date.now() - 48 * 60 * 60 * 1000)
    
    // Find unpurchased designs older than 48 hours
    const { data: expiredDesigns, error } = await supabase
      .from('digital_products')
      .select('*')
      .eq('status', 'unpurchased')
      .lt('created_at', fortyEightHoursAgo.toISOString())
    
    if (error) throw error
    
    let cleanedCount = 0
    
    for (const design of expiredDesigns || []) {
      // Delete from storage
      if (design.download_url) {
        const filePath = design.download_url.split('/').pop()
        await supabase.storage
          .from('digital-products')
          .remove([`${design.user_id}/${design.type}/${filePath}`])
      }
      
      // Delete from database
      await supabase
        .from('digital_products')
        .delete()
        .eq('id', design.id)
      
      cleanedCount++
    }
    
    return NextResponse.json({ 
      success: true, 
      message: `Cleaned up ${cleanedCount} expired designs` 
    })
  } catch (error) {
    return NextResponse.json({ error: error.message }, { status: 500 })
  }
}
```

export default function DigitalPrintingPage() {

    // Load design data from sessionStorage
    const pendingDesign = sessionStorage.getItem('pendingDesignForPrint')
    if (pendingDesign) {
      setDesignData(JSON.parse(pendingDesign))
    }
  }, [])
  
  const handleAddToCart = () => {
    if (!designData || !selectedMaterial || !selectedSize) return
    
    const printItem = {
      id: `print-${designData.id}`,
      name: `${designData.name} - ${selectedMaterial} Print`,
      type: 'print',
      price: calculatePrintPrice(selectedMaterial, selectedSize, quantity),

        designId: designData.id,
        designUrl: designData.previewUrl,
        material: selectedMaterial,
        size: selectedSize,
        generationInputs: designData.generationInputs
      }
    }
    
    // Add to regular cart
    addToCart(printItem)
    
    // Clear session storage
    sessionStorage.removeItem('pendingDesignForPrint')

    router.push('/cart')
  }

              src={designData.previewUrl} 
              alt={designData.name}
              className="w-full max-w-md mx-auto rounded-lg border"
            />
            <p className="text-gray-600">{designData.name}</p>
